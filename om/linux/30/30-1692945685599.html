                              # 环境准备

                              # 检查系统是否安装jdk

                              rpm -qa | grep java

                              # 卸载预装

                              rpm -e --nodeps java-1.8.0-openjdk-headless-1.8.0.181-7.b13.el7.x86_64
#相同的命令 删除上图中与openjdk相关的软件即可

                              # 下载JDK安装包

                              前往Oracle官网下载JDK8的安装包。

                              链接：https://www.oracle.com/java/technologies/javase/javase-jdk8-downloads.htmlopen in new window

                              备用链接:

                              阿里云盘：https://www.aliyundrive.com/s/scCYi5ZZw5Qopen in new window

                              天翼云盘：https://cloud.189.cn/web/share?code=eamqui63i6baopen in new window

                              百度网盘：https://pan.baidu.com/s/1Rzh66EPELy2u5otGtAZI6Aopen in new window 提取码：java

                              # 解压缩

                              su
mkdir /usr/local/java
cd /usr/local/java
tar -zxf /home/soft01/jdk1.8.0_271.tar.gz

                              # 创建脚本文件

                              cd /etc
cp profile profile.bak
vim profile

                              并添加如下内容(版本号可能会不同,注意修改对应的版本号)

                              export PATH=/usr/local/java/jdk1.8.0_271/bin:$PATH
export JAVA_HOME=/usr/local/java/jdk1.8.0_271
export CLASSPATH=.

                              # 使脚本生效

                              执行如下命令,使脚本文件生效

                              source profile

                              # 验证

                              java -version
