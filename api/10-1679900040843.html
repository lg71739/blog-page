    # 获取盘点任务列表

    获取系统中的 盘点任务列表

    # 接口地址

    /api/resinventory/getList

    # 接口参数

    无

    # 接口Data数据说明

    参数名格式参数说明示例值
    idString盘点任务ID1641710025134518273
    nameString盘点任务名称2022年资产盘点
    batchidString盘点单批次号
    statusString盘点状态

    - wait 待盘点
    - doing 盘点中
    - finish 已盘点
    - cancel 取消    		wait
    rescatnameArray资产分类名称doing
    starttimeDate资产盘点开始时间
    finishtimeDate资产盘点结束时间

    # 获取盘点任务详情

    获取某一盘点任务下的所有的资产列表

    # 接口请求地址

    /api/resinventory/getItem

    # 接口请求参数

    参数名参数说明参考值
    pdid盘点任务列表ID1641713297307316225

    # 接口返回数据

    参数名参数格式参数说明示例值
    idString接口状态码。200为正常，其余为异常状态200
    pdidArray盘点任务中设备列表
    resnameString状态信息。当接口异常时，返回异常信息
    residString资产ID
    uuidString资产编号(与扫描的标签code匹配)
    pdstatusString盘点状态( wait | finish )

    # 接口返回示例

    "success": true,
    "code": "200",
    "message": "请求成功",

    # 更新盘点物资状态

    将扫描到的资产置为已盘点状态

    # 接口地址

    /api/resinventory/upDateItemState

    # 接口参数

    参数名参数说明参考值
    itemId扫描到code匹配的物资IDbfde8cb3-b671-45fe-9c4c-6e481c8b988a

    # 接口返回示例

    {
    "success": true,
    "code": "200",
    "message": "请求成功"
}
